CREATE TABLE "CSMS_DETAILS" (
	"ID"	INTEGER,
	"VERSION"	TEXT NOT NULL,
	"UPDATE_NAME"	TEXT NOT NULL UNIQUE,
	"UPDATE_ON"	TEXT,
	PRIMARY KEY("ID" AUTOINCREMENT)
);

CREATE TABLE "USER" (
	"USER_ID"	INTEGER,
	"NAME"	TEXT NOT NULL,
	"PHONE"	INTEGER NOT NULL,
	"EMAIL"	TEXT NOT NULL UNIQUE,
	"PASSWORD"	TEXT NOT NULL,
	"USER_TYPE"	TEXT NOT NULL DEFAULT 'N',
	PRIMARY KEY("USER_ID" AUTOINCREMENT)
);

CREATE TABLE "STUDENT" (
	"ROLL_NO"	TEXT,
	"NAME"	TEXT NOT NULL,
	"COURSE"	TEXT NOT NULL,
	"BATCH"	TEXT NOT NULL,
	"DOB"	TEXT NOT NULL,
	"GENDER"	TEXT NOT NULL,
	"BLOOD_GROUP"	TEXT,
	"PHONE"	TEXT NOT NULL,
	"EMAIL"	TEXT,
	"ADDRESS"	TEXT,
	"FATHER_NAME"	TEXT,
	"MOTHER_NAME"	TEXT,
	"ADDDATE"	TEXT NOT NULL,
	PRIMARY KEY("ROLL_NO")
);

CREATE TABLE "MASTER_BOOK" (
	"MSTBK_ID"	INTEGER,
	"BOOK_NAME"	TEXT NOT NULL,
	"AUTHOR"	TEXT NOT NULL,
	"PUBLICATION"	TEXT NOT NULL,
	"PRICE"	INTEGER NOT NULL,
	PRIMARY KEY("MSTBK_ID" AUTOINCREMENT)
);

CREATE TABLE "CHILD_BOOK" (
	"MSTBK_ID"	INTEGER NOT NULL,
	"CHLBK_ID"	TEXT,
	"REMARKS"	TEXT,
	"BK_ADDDATE"	INTEGER,
	"BK_STATUS"	TEXT NOT NULL DEFAULT 'N',
	"ROLL_NO"	TEXT,
	"FROM"	TEXT,
	"TO"	TEXT,
	PRIMARY KEY("CHLBK_ID")
);

CREATE TABLE "TRANSACTION" (
	"TRANS_ID"	INTEGER,
	"MSTBK_ID"	INTEGER NOT NULL,
	"CHLBK_ID"	TEXT NOT NULL,
	"ROLL_NO"	TEXT NOT NULL,
	"FROM"	TEXT NOT NULL,
	"TO"	TEXT NOT NULL,
	"GIVEN_ON"	TEXT NOT NULL,
	"REMARKS"	TEXT NOT NULL,
	PRIMARY KEY("TRANS_ID" AUTOINCREMENT)
);
